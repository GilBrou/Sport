import axios from 'axios';

const instance = axios.create({
	baseURL: 'http://localhost:3000/user',
});

/**
 * Get user infos from the API
 */
export const getUserInfos = async (id) => {
	try {
		const res = await instance.get(`/${id}`);
		return res.data;
	} catch (e) {
		console.log(e);
	}
};
